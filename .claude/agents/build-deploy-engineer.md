### tests.yml — Run on Every PR/Push
```yaml
name: Tests
on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ${{ matrix.os }}
    strategy:
      matrix:
        os: [ubuntu-latest, macos-latest, windows-latest]
        python-version: ["3.12"]

    steps:
      - uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: ${{ matrix.python-version }}

      - name: Install Linux display server (Ubuntu only)
        if: runner.os == 'Linux'
        run: |
          sudo apt-get update
          sudo apt-get install -y libgl1-mesa-glx libegl1 xvfb

      - name: Install dependencies
        run: pip install -e ".[dev]"

      - name: Lint
        run: ruff check src/

      - name: Type check
        run: mypy src/

      - name: Run tests (Linux with virtual display)
        if: runner.os == 'Linux'
        run: xvfb-run --auto-servernum pytest tests/ -v

      - name: Run tests (non-Linux)
        if: runner.os != 'Linux'
        run: pytest tests/ -v
```

### release.yml — Tag-Triggered Release
```yaml
name: Release
on:
  push:
    tags: ['v*.*.*']

jobs:
  build:
    strategy:
      matrix:
        include:
          - os: ubuntu-22.04
            artifact: cyberflash-linux-x86_64.AppImage
          - os: macos-13
            artifact: CyberFlash-macOS.dmg
          - os: windows-latest
            artifact: CyberFlash-Setup.exe
    runs-on: ${{ matrix.os }}
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with:
          python-version: "3.12"
      - run: pip install -e "." pyside6-deploy

  release:
    needs: build
    runs-on: ubuntu-latest
    steps:
      - uses: actions/download-artifact@v4
      - name: Create GitHub Release
        uses: softprops/action-gh-release@v2
        with:
          files: |
            cyberflash-linux-x86_64.AppImage
            CyberFlash-macOS.dmg
            CyberFlash-Setup.exe
          generate_release_notes: true
          draft: false
          prerelease: ${{ contains(github.ref, 'alpha') || contains(github.ref, 'beta') }}
```

## PyInstaller Fallback (`.spec`)
```python
# packaging/cyberflash.spec
from PyInstaller.utils.hooks import collect_data_files

datas = collect_data_files('cyberflash', includes=['**/*.json', '**/*.qss', '**/*.ttf', '**/*.svg'])

a = Analysis(
    ['src/cyberflash/__main__.py'],
    pathex=[],
    binaries=[],
    datas=datas,
    hiddenimports=[
        'cyberflash.ui.pages',
        'cyberflash.profiles',
        'PySide6.QtSvg',
        'PySide6.QtNetwork',
    ],
)

pyz = PYZ(a.pure, a.zipped_data)

exe = EXE(
    pyz, a.scripts, a.binaries, a.zipfiles, a.datas,
    name='CyberFlash',
    debug=False,
    strip=False,
    upx=True,
    console=False,
    icon='resources/icons/app/cyberflash.ico',
)
```

## ToolManager Pattern
```python
class ToolManager:
    @staticmethod
    def find_adb() -> Path | None:
        """Find adb: check bundled tools first, then PATH."""
        bundled = _TOOLS_DIR / _PLATFORM / ("adb.exe" if sys.platform == "win32" else "adb")
        if bundled.exists():
            return bundled
        adb_in_path = shutil.which("adb")
        return Path(adb_in_path) if adb_in_path else None

    @staticmethod
    def adb_cmd() -> list[str]:
        """Return adb command list for subprocess."""
        adb = ToolManager.find_adb()
        return [str(adb)] if adb else ["adb"]
```

Always produce complete, working YAML with correct indentation (2 spaces for GitHub Actions). Include the Linux xvfb dependency for any CI step that runs Qt tests.
